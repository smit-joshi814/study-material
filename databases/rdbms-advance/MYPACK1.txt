CREATE OR REPLACE PACKAGE pypack1 IS
PROCEDURE TRIANGLE(N NUMBER);
END mypack;

CREATE OR REPLACE PACKAGE BODY mypack1 IS
PROCEDURE TRIANGLE(N NUMBER) IS
I NUMBER;
J NUMBER;
BEGIN
	FOR I IN 1..N
	LOOP
		FOR J IN 1..I
		LOOP
			DBMS_OUTPUT.PUT('*');
		END LOOP;
	DBMS_OUTPUT.PUT_LINE(' ');
	END LOOP;

	FOR I IN REVERSE 1..N-1
	LOOP
		FOR J IN 1..I
		LOOP
			DBMS_OUTPUT.PUT('*');
		END LOOP;
	DBMS_OUTPUT.PUT_LINE(' ');
	END LOOP;
END TRIANGLE;
end mypack1;

DECLARE 
N NUMBER;
BEGIN
N:=&ENTER_NUMBER;
MYPACK1.TRIANGLE(N);
END;
