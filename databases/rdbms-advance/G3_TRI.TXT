CREATE OR REPLACE TRIGGER CHK_NAME 
BEFORE INSERT OR UPDATE ON CUSTOMER
FOR EACH ROW
DECLARE
I NUMBER;
NAME VARCHAR2(10);
CH VARCHAR(1);

COUNTER NUMBER:=0;
BEGIN
NAME:=(:NEW.CNAME);
NAME:=UPPER(NAME);
	FOR I IN 1..LENGTH(NAME)
	LOOP
	CH:=SUBSTR(NAME,I,1);
		IF ( (ASCII(CH) >=65 AND ASCII(CH) <= 90) OR ASCII(CH)=32 ) THEN
			COUNTER:=COUNTER+1;	
		END IF;
	END LOOP;
	IF(COUNTER = LENGTH(NAME)) THEN
		DBMS_OUTPUT.PUT_LINE('NAME IS VALID');
	ELSE
		RAISE_APPLICATION_ERROR(-20780,'PLEASE ENTER VALID NAME');
	END IF;
END CHK_NAME;
/
