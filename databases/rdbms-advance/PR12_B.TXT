CREATE OR REPLACE PROCEDURE GetStatus(Obj VARCHAR2,ObjOwner VARCHAR2) IS
CURSOR C1 IS
	SELECT STATUS,OBJECT_NAME FROM DBA_OBJECTS WHERE OBJECT_TYPE=Obj AND OWNER=ObjOwner;
REC C1%ROWTYPE;
BEGIN
	OPEN C1;
	DBMS_OUTPUT.PUT_LINE('STATUS');
	LOOP
	FETCH C1 INTO REC;
	EXIT WHEN C1%NOTFOUND;
	DBMS_OUTPUT.PUT_LINE(REC.OBJECT_NAME||'	'||REC.STATUS);
	END LOOP;
CLOSE C1;
END GetStatus;
/
