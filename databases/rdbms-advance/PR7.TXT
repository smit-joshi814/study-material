CREATE OR REPLACE FUNCTION GetTBLspaceName(tNAME VARCHAR2) RETURN NUMBER IS
CURSOR C1 IS
        SELECT OWNER,TABLESPACE_NAME FROM DBA_TABLES WHERE TABLE_NAME=tNAME;
REC C1%ROWTYPE;
BEGIN
        OPEN C1;
        DBMS_OUTPUT.PUT_LINE('OWNER             TABLESPACE_NAME');
        DBMS_OUTPUT.PUT_LINE('---------------------------');
        LOOP
        FETCH C1 INTO REC;
        EXIT WHEN C1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(REC.OWNER||'       '||REC.TABLESPACE_NAME);
        END LOOP;
        CLOSE C1;
RETURN 0;
END GetTBLspaceName;
/
