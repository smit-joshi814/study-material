DECLARE
I NUMBER;
NAME VARCHAR2(10);
CH VARCHAR(1);
N NUMBER;
COUNTER NUMBER:=0;
BEGIN
NAME:='&NAME';
NAME:=UPPER(NAME);
	FOR I IN 1..LENGTH(NAME)
	LOOP
	CH:=SUBSTR(NAME,I,1);
	IF ( ASCII(CH) >=65 AND ASCII(CH) <= 90 ) THEN
		COUNTER:=COUNTER+1;	
	END IF;
	END LOOP;
	IF(COUNTER = LENGTH(NAME)) THEN
	DBMS_OUTPUT.PUT_LINE('NAME IS VALID');
	ELSE
	DBMS_OUTPUT.PUT_LINE('PLEASE ENTER VALID NAME');
	END IF;
END;
