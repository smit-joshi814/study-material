CREATE OR REPLACE TRIGGER LOG_OUT_INFO
BEFORE LOGOFF ON DATABASE
BEGIN
INSERT INTO LOG_IN_OUT_INFO VALUES(SYS.LOGIN_USER,SYS.SYSEVENT,SYSDATE,TO_CHAR(SYSDATE,'HH24:MI:SS'));
END LOG_OUT_INFO;


CREATE OR REPLACE TRIGGER LOG_IN_INFO
AFTER LOGON ON DATABASE
BEGIN
INSERT INTO LOG_IN_OUT_INFO VALUES(SYS.LOGIN_USER,SYS.SYSEVENT,SYSDATE,TO_CHAR(SYSDATE,'HH24:MI:SS'));
END LOG_IN_INFO;
