CREATE OR REPLACE PROCEDURE GetIndex (tNAME VARCHAR2) IS
CURSOR C1 IS 
	SELECT OWNER,INDEX_NAME,INDEX_TYPE FROM DBA_INDEXES WHERE TABLE_NAME=tNAME;
REC C1%ROWTYPE;
BEGIN
	OPEN C1;
	DBMS_OUTPUT.PUT_LINE('OWNER		INDEX_NAME			INDEX_TYPE');
	DBMS_OUTPUT.PUT_LINE('-------------------------------------------');
	LOOP
	FETCH C1 INTO REC;
	EXIT WHEN C1%NOTFOUND;
	DBMS_OUTPUT.PUT_LINE(	REC.OWNER||'	'||REC.INDEX_NAME||'		'||REC.INDEX_TYPE);
	END LOOP;
	CLOSE C1;
END GetIndex;
/
