DECLARE
XSNUM NUMBER(4,2);
RESULT VARCHAR(100);
BEGIN
XSNUM:=&SNUM;
SELECT TO_CHAR(TO_DATE(TRUNC(AMOUNT),'J'),'JSP')||' INR RUPEES AND '||
TO_CHAR(TO_DATE(SUBSTR(AMOUNT,INSTR(AMOUNT,'.')+1),'J'),'JSP') ||' PAIESE'  INTO RESULT FROM ORDERS WHERE SNUM=XSNUM;
DBMS_OUTPUT.PUT_LINE(RESULT);
END;
/
