CREATE OR REPLACE TRIGGER CHK_NAME
BEFORE INSERT OR UPDATE ON CUSTOMER
FOR EACH ROW 
DECLARE
I NUMBER;
C NUMBER;
CH VARCHAR2;
BEGIN
	FOR I IN 1..LENGTH(:NEW.CNAME)
	LOOP

		:NEW.CNAME:=(UPPER(:NEW.CNAME));	--TO UPPERCASE
		CH:=SUBSTR(:NEW.CNAME,I,1);

		IF(CH IN('A','E','I','O','U')) THEN
			C:=C+1;
		END IF;

	END LOOP;

	IF(C=0) THEN

		RAISE_APPLICATION_ERROR(-20611,'INCORRECT NAME');

	END IF;

END CHK_NAME;
