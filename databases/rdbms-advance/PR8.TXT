CREATE OR REPLACE PROCEDURE GetProName(pOWNER IN VARCHAR2) IS
    CURSOR C1 IS
        SELECT OBJECT_NAME,STATUS,LAST_DDL_TIME FROM DBA_OBJECTS WHERE OWNER=pOWNER AND OBJECT_TYPE='PROCEDURE';
    REC C1%ROWTYPE;
    BEGIN
       OPEN C1;
       DBMS_OUTPUT.PUT_LINE('OBJECT_NAME               STATUS  LAST_DDL_TIME');
       DBMS_OUTPUT.PUT_LINE('----------------------------------------');
       LOOP
       FETCH C1 INTO REC;
       EXIT WHEN C1%NOTFOUND;
       DBMS_OUTPUT.PUT_LINE(REC.OBJECT_NAME||'      '||REC.STATUS||'     '||REC.LAST_DDL_TIME);
       END LOOP;
CLOSE C1;
END GetProName;
